Base marcas
1- clasifica los productos por su precio
 10000 “muy costoso”
 >5000 “costoso”
 resto “Normal”
2- Calificar los productos según su calificación:
 mayor a 8 "Muy BUENO"
 mayor a 6 "BUENO"
 resto "REGULAR":


1)
select nombre, precio,
case
when precio >=10000 then 'Muy costoso'
when precio >=5000 then 'Costoso'
else 'Normal'
end as clasificacion
from productos;

2)
select nombre, puntuacion,
case
when puntuacion >8 then 'Muy Bueno'
when puntuacion >6 then 'Bueno'
else 'Regular'
end as aceptacion
from productos;

3)
SELECT apellido, salario, oficio, fecha_alt, dept_No,
case 
when dept_No = 10 then 'rescindido'
when dept_No = 30 then 'rescidido'
ELSE  'en actividad'
END AS 'situacion laboral'
FROM emp;

4)
SELECT apellido, salario,
case
when salario<100000 then 'Subir el sueldo'
when salario>=100000 AND salario <=250000 then 'Mantener el sueldo'
ELSE 'Bajar el sueldo'
END AS 'Situacion salarial'
FROM emp
UNION ALL  
SELECT apellido, salario,
case
when salario<100000 then 'Subir el sueldo'
when salario>=100000 AND salario <=250000 then 'Mantener el sueldo'
ELSE 'Bajar el sueldo'
END AS 'Situacion salarial'
FROM plantilla;

6)
SELECT Apellido, Direccion, Fecha_Nac, S,
case
when S = 'F' then 'Femenino'
when S = 'M' then 'Masculino' 
END AS 'Sexo',
case
when year(STR_TO_DATE(Fecha_Nac, '%d-%M-%Y')) BETWEEN 1940 AND 1949 then 'Década del 40'
when year(STR_TO_DATE(Fecha_Nac, '%d-%M-%Y')) BETWEEN 1950 AND 1959 then 'Década del 50'
when year(STR_TO_DATE(Fecha_Nac, '%d-%M-%Y')) BETWEEN 1960 AND 1969 then 'Década del 60'
when year(STR_TO_DATE(Fecha_Nac, '%d-%M-%Y')) BETWEEN 1970 AND 1979 then 'Década del 70'
ELSE 'Década del 80'
END AS 'Década'
FROM enfermo; 
 
o también si se considera la década de la edad:
SELECT Apellido, Direccion, Fecha_Nac, S,
case
when S = 'F' then 'Femenino'
when S = 'M' then 'Masculino' 
END as 'Sexo',
case
when FLOOR(DATEDIFF(CURDATE(), STR_TO_DATE(Fecha_Nac, '%d-%M-%Y'))/365)>=80 then 'Está en la década de los 80'
when FLOOR(DATEDIFF(CURDATE(), STR_TO_DATE(Fecha_Nac, '%d-%M-%Y'))/365)>=70 then 'Está en la década de los 70'
when FLOOR(DATEDIFF(CURDATE(), STR_TO_DATE(Fecha_Nac, '%d-%M-%Y'))/365)>=60 then 'Está en la década de los 60'
when FLOOR(DATEDIFF(CURDATE(), STR_TO_DATE(Fecha_Nac, '%d-%M-%Y'))/365)>=50 then 'Está en la década de los 50'
when FLOOR(DATEDIFF(CURDATE(), STR_TO_DATE(Fecha_Nac, '%d-%M-%Y'))/365)>=40 then 'Está en la década de los 40'
ELSE 'Está en una década menor a los 40'
END AS 'Década de vida'
FROM enfermo;

7)
SELECT A.Apellido, A.Salario, A.Oficio, 
case 
when A.Comision=0 then 'Sin comisión'
ELSE 'Con comisión'
END AS 'Comisión',
case 
when A.Dept_No IS NOT NULL then E.DNombre
ELSE 'No pertenece a ningún departamento'
END AS 'Departamento'
FROM emp AS A
LEFT JOIN dept AS E ON A.Dept_No=E.Dept_No;



o también:
SELECT dept.DNombre, emp.Apellido, emp.Salario, emp.Oficio,
case
when dept.DNombre IS NULL then 'No pertenece a nigún departamento'
ELSE dept.DNombre
END AS 'Departamento',
case
when emp.Comision=0 then 'Sin comision'
ELSE 'Con comision'
END AS 'Comisión'
FROM emp
LEFT JOIN dept ON emp.Dept_No=dept.Dept_No;

8)
SELECT sala.Num_cama, hospital.Nombre AS 'hospital', sala.Nombre AS 'Sala', floor((sala.Num_cama*100)/hospital.Num_cama) as 'porcentaje de camas',
case 
when sala.Num_Cama=0 then 'No tiene camas'
when sala.Num_Cama>=90 then 'Tiene muchas camas'
when sala.Num_Cama>=40 then 'Buen número'
else 'Tiene camas'
end as 'clasificacion de camas'
FROM hospital 
INNER JOIN sala ON hospital.Hospital_cod=sala.Hospital_cod;

Base contactos:

9)
SELECT apellidos, nombre, localidad,
if (localidad = 'Barcelona', 'Cambia de localidad', 'No cambia de localidad')
FROM contactos;

ó también se puede:
SELECT apellidos, nombre, localidad,
case
when localidad = 'Barcelona' then 'Cambia de localidad'
ELSE 'No cambia de localidad'
END AS 'Relocalización'
FROM contactos;

10)
SELECT apellidos, nombre, sexo, altura,
case 
when sexo='M' AND altura>=1.75 then 'Hombre alto'
ELSE 'No es alto'
END AS 'clasificación altura'
FROM contactos;
